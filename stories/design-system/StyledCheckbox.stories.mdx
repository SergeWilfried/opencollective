import { Meta, Story, Canvas } from '@storybook/addon-docs/blocks';
import StyledCheckbox from '../../components/StyledCheckbox';

<Meta title="Design system/StyledCheckbox" component={StyledCheckbox} />

### Normal

<Canvas>
  <Story name="Normal">
    {() => {
      const [checked, setChecked] = React.useState(false);
      return (
        <StyledCheckbox
          checked={checked}
          onChange={({ checked }) => setChecked(checked)}
          label={checked ? 'Checked' : 'Unchecked'}
        />
      );
    }}
  </Story>
</Canvas>

### Disabled

<Canvas>
  <Story name="Disabled">
    {() => {
      return (
        <div>
          <StyledCheckbox disabled label="Unchecked" defaultChecked={false} />
          <br />
          <StyledCheckbox disabled label="Checked" defaultChecked />
        </div>
      );
    }}
  </Story>
</Canvas>

### Without label

<Canvas>
  <Story name="Without Label">
    {() => {
      return <StyledCheckbox />;
    }}
  </Story>
</Canvas>

### With custom sizes

<Canvas>
  <Story name="With Custom Sizes">
    {() => {
      const [checked, setChecked] = React.useState(false);
      return (
        <div>
          <StyledCheckbox
            label="A little bit bigger"
            size="20px"

            checked={checked}
            onChange={({ checked }) => setChecked(checked)}
          />
        </div>
      );
    }}
  </Story>
</Canvas>
