import { Story, Canvas, Meta } from '@storybook/addon-docs/blocks';
import ConfirmationModal from '../../components/ConfirmationModal';
import StyledButton from '../../components/StyledButton';

<Meta title="Design system/ConfirmationModal" component={ConfirmationModal} />

<Canvas>
  <Story name="Default">
    {() => {
      const [show, setShow] = React.useState(false);
      return (
        <div>
          {show ? (
            <ConfirmationModal
              header={'Use modal'}
              body={'Are you sure you want to use this modal?'}
              onClose={() => setShow(false)}
              cancelLabel={'No'}
              cancelHandler={() => setShow(false)}
              continueLabel={'Yes'}
              continueHandler={console.log}
            />
          ) : (
            <StyledButton buttonSize="large" buttonStyle="primary" onClick={() => setShow(true)}>
              Show modal
            </StyledButton>
          )}
        </div>
      );
    }}
  </Story>
</Canvas>
