export const getActionsComponent = actions => {
  if (actions === 'Example button') {
    return <NotificationBarButton>Send $18.33 to Agora collective</NotificationBarButton>;
  } else if (actions === 'Example link') {
    return <NotificationBarLink href="#">Go back</NotificationBarLink>;
  }
  return null;
};
