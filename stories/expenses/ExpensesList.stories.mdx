import { ArgsTable, Meta, Story, Canvas } from '@storybook/addon-docs/blocks';
import ExpensesList from '../../components/expenses/ExpensesList';
import { expensesList } from '../mocks/expenses-v2';
import { webpackCollective } from '../mocks/collectives';

<Meta title="Expenses/ExpensesList" component={ExpensesList} />

export const DefaultStory = props => <ExpensesList collective={webpackCollective} expenses={expensesList} />;

<Story name="Default">{DefaultStory.bind({})}</Story>

<ArgsTable story="Default" />

    {() => {
      const [isLoading, setLoading] = React.useState(false);
      const [isInverted, setInverted] = React.useState(false);
      return (
        <div>
          <button onClick={() => setLoading(!isLoading)}>Toggle loading</button>
          <button onClick={() => setInverted(!isInverted)}>Toggle inverted</button>
          <br />
          <hr />
          <br />
          <ExpensesList
            collective={webpackCollective}
            expenses={expensesList}
            isLoading={isLoading}
            isInverted={isInverted}
          />
        </div>
      );
    }}
  </Story>
</Canvas>
